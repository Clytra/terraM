get user by id

url = "{{url}}/api/user/1"

payload = "{\n\t\"username\": \"Test01\",\n\t\"password\": \"qwerty\"\n}"
headers = {
  'Content-Type': 'application/json'
}

response = requests.request("GET", url, headers=headers, data = payload)
